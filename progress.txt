── HOW TO RUN ───────────────────────────────────────────────

cd code
docker compose up -d                      # start DB
cd apps/backend && npm run start:dev      # backend :3001
cd apps/frontend && npm run dev           # frontend :3000

# AI config (code/apps/backend/.env)
AI_PROVIDER=openrouter                    # or: anthropic
AI_API_KEY=sk-or-...                      # your key
AI_MODEL=google/gemini-2.5-flash-preview  # or any OpenRouter model

# Tests
cd apps/backend && npm test               # unit tests (no DB needed)
cd apps/backend && npm run test:e2e       # e2e tests (Docker DB required)
cd apps/frontend && npm test              # frontend unit tests
